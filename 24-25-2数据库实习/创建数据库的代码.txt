CREATE TABLE students (
    sid INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    sex ENUM('m', 'f') NOT NULL,
    class INT NOT NULL,
    major VARCHAR(100) NOT NULL
);

create table teachers(
    tid int UNSIGNED auto_increment primary key,
    name varchar(100) not null,
    sex enum('m','f') not null
);

create table clubs(
    cid int UNSIGNED auto_increment primary key,
    name varchar(100)
);

create table events(
    eid int UNSIGNED auto_increment primary key,
    name varchar(100) not null,
    place varchar(100) not null,
    cid int not null,
    date date not null
);

create table tc(
    tid int  UNSIGNED not null,
    cid int  UNSIGNED not null,
    primary key(tid, cid),
    foreign key(tid) references teachers(tid),
    foreign key(cid) references clubs(cid)
);

create table sc(
    sid int UNSIGNED not null,
    cid int UNSIGNED not null,
    primary key(sid, cid),
    foreign key(sid) references students(sid),
    foreign key(cid) references clubs(cid)
);

create table se(
    sid int UNSIGNED not null,
    eid int UNSIGNED not null,
    primary key(sid, eid),
    foreign key(sid) references students(sid),
    foreign key(eid) references events(eid)
);
